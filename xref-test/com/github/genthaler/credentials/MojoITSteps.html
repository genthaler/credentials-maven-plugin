23  import static org.hamcrest.Matchers.containsString;
24  import static org.hamcrest.Matchers.not;
25  import static org.hamcrest.Matchers.nullValue;
26  import static org.junit.Assert.assertThat;
27  
28  import java.io.IOException;
29  import java.util.Arrays;
30  
31  import org.apache.maven.it.VerificationException;
32  import org.apache.maven.it.Verifier;
33  import org.codehaus.plexus.util.xml.Xpp3Dom;
34  import org.codehaus.plexus.util.xml.pull.XmlPullParserException;
35  
36  import cucumber.api.java.After;
37  import cucumber.api.java.Before;
38  import cucumber.api.java.en.Given;
39  import cucumber.api.java.en.Then;
40  import cucumber.api.java.en.When;
41  
42  /**
43   * Integration test for SetMojo and SetAllMojo.
44   */
45  public class MojoITSteps {
46  
47  	private Throwable thrown = null;
48  	private MojoITState state;
49  	private Verifier verifier;
50  	private static int folder = 0;
51  
52  	@Before
53  	public void setUp() throws VerificationException, IOException,
54  			XmlPullParserException {
55  		state = new MojoITState(folder++);
56  
57  		verifier = new Verifier(state.getExecDir().getAbsolutePath());
58  		verifier.setLocalRepo(state.getLocalRepository().getAbsolutePath());
59  		verifier.setAutoclean(false);
60  		// verifier.setForkJvm(false);
61  		thrown = null;
62  	}
63  
64  	@Given("^no credentials plugin configuration$")
65  	public void emptyMojo() {
66  		state.getModel().getBuild().getPlugins()
67  				.remove(state.getCredentialsPlugin());
68  	}
69  
70  	@SuppressWarnings("serial")
71  	@Given("^the credentials-maven-plugin (.*) property is (.*)$")
72  	public void setMojoProperty(String property, final String propertyValue) {
73  		Xpp3Dom configuration = (Xpp3Dom) state.getCredentialsPlugin()
74  				.getConfiguration();
75  		configuration.addChild(new Xpp3Dom(property) {
76  			{
77  				setValue(propertyValue);
78  			}
79  		});
80  	}
81  
82  	@Given("^the (.*) Server username is (.*)$")
83  	public void setServerUsername(String id, String value) {
84  		state.getServer(id).setUsername(value);
85  	}
86  
87  	@Given("^the (.*) Server password is (.*)$")
88  	public void setServerPassword(String id, String value) {
89  		state.getServer(id).setPassword(value);
90  	}
91  
92  	@Given("^the (.*) commandline property is (.*)$")
93  	public void setSystemProperty(String property, String value) {
94  		verifier.setSystemProperty(property, value);
95  	}
96  
97  	@When("^the (.*) goal is executed")
98  	public void executeMojo(String goal) throws VerificationException,
99  			IOException {
100 		try {
101 			state.writeState();
102 			verifier.addCliOption(String.format("-s %s", state
103 					.getSettingsFile().getAbsolutePath()));
104 
105 			verifier.executeGoals(Arrays.asList(state.getGroupId() + ":"
106 					+ state.getArtifactId() + ":" + state.getVersion() + ":"
107 					+ goal, MojoITState.SCRIPT_GROUP_ID + ":"
108 					+ MojoITState.SCRIPT_ARTIFACT_ID + ":"
109 					+ MojoITState.SCRIPT_VERSION + ":execute"));
110 		} catch (VerificationException e) {
111 			thrown = e;
112 		} catch (IOException e) {
113 			thrown = e;
114 		}
115 	}
116 
117 	@Then("^there should have been a Project (.*) property with value (.*)$")
118 	public void assertProjectProperty(String property, String value)
119 			throws VerificationException {
120 		assertThat("An exception was thrown", thrown, nullValue());
121 		verifier.verifyTextInLog("project." + property + "=" + value);
122 	}
123 
124 	@Then("^there should have been a System (.*) property with value (.*)$")
125 	public void assertSystemProperty(String property, String value)
126 			throws VerificationException {
127 		assertThat("An exception was thrown", thrown, nullValue());
128 		verifier.verifyTextInLog("system." + property + "=" + value);
129 	}
130 
131 	@Then("^there should have been no errors$")
132 	public void noErrors() throws VerificationException {
133 		// try {
134 		assertThat("An exception was thrown", thrown, nullValue());
135 		verifier.verifyErrorFreeLog();
136 		// } catch (VerificationException e) {
137 		// e.printStackTrace();
138 		// }
139 	}
140 
141 	@Then("^an exception should be thrown with the message '(.*)'$")
142 	public void anExceptionIsThrown(String message)
143 			throws VerificationException {
144 		assertThat("It looks like there was no exception thrown", thrown,
145 				not(nullValue()));
146 
147 		assertThat(
148 				"Thrown exception doesn't contain the expected message fragment",
149 				thrown.getMessage(), containsString(message));
150 	}
151 
152 	@After
153 	public void tearDown() throws Exception {
154 		verifier.resetStreams();
155 	}
156 }
