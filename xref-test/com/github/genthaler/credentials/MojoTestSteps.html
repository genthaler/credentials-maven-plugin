23  import java.util.ArrayList;
24  import java.util.List;
25  import java.util.Properties;

27  import junit.framework.AssertionFailedError;
28  
29  import org.apache.maven.plugin.MojoExecutionException;
30  import org.apache.maven.plugin.MojoFailureException;
31  import org.apache.maven.plugin.logging.SystemStreamLog;
32  import org.apache.maven.plugin.testing.AbstractMojoTestCase;
33  import org.apache.maven.plugin.testing.stubs.MavenProjectStub;
34  import org.apache.maven.settings.Server;
35  import org.apache.maven.settings.Settings;
36  import org.sonatype.plexus.components.sec.dispatcher.SecDispatcher;
37  
38  import com.github.genthaler.credentials.AbstractCredentialsMojo;
39  import com.github.genthaler.credentials.SetAllMojo;
40  import com.github.genthaler.credentials.SetMojo;
41  
42  import cucumber.api.java.After;
43  import cucumber.api.java.Before;
44  import cucumber.api.java.en.Given;
45  import cucumber.api.java.en.Then;
46  import cucumber.api.java.en.When;
47  
48  /**
49   * Unit test for SetMojo.
50   */
51  public class MojoTestSteps extends AbstractMojoTestCase {
52  	private AbstractCredentialsMojo mojo;
53  	private MavenProjectStub project;
54  	private Settings settings;
55  	private Properties properties;
56  	private Throwable thrown = null;
57  	private List<String> systemProperties;
58  
59  	@Before(value = { "@set" }, order = 1)
60  	public void setUpSetMojo() {
61  		mojo = new SetMojo();
62  	}
63  
64  	@Before(value = { "@set-all" }, order = 1)
65  	public void setUpSetAllMojo() {
66  		mojo = new SetAllMojo();
67  	}
68  
69  	@Override
70  	@Before(order = 2)
71  	public void setUp() throws Exception {
72  		super.setUp();
73  		SecDispatcher securityDispatcher = (SecDispatcher) lookup(
74  				"org.sonatype.plexus.components.sec.dispatcher.SecDispatcher",
75  				"default");
76  		setVariableValueToObject(mojo, "securityDispatcher", securityDispatcher);
77  		securityDispatcher.decrypt("");
78  		properties = new Properties();
79  		project = new MavenProjectStub() {
80  			@Override
81  			public Properties getProperties() {
82  				return properties;
83  			}
84  		};
85  		setVariableValueToObject(mojo, "project", project);
86  
87  		settings = new Settings();
88  		setVariableValueToObject(mojo, "settings", settings);
89  		systemProperties = new ArrayList<String>();
90  
91  		mojo.setLog(new SystemStreamLog() {
92  			@Override
93  			public boolean isDebugEnabled() {
94  				return true;
95  			}
96  		});
97  	}
98  
99  	@Given("^a (set.*) Plugin")
100 	public void emptyMojo(String goal) {
101 	}
102 
103 	@Given("^the Plugin (.*) property is (.*)$")
104 	public void setMojoProperty(String property, String value)
105 			throws IllegalAccessException {
106 		if ("useSystemProperties".equalsIgnoreCase(property))
107 			setVariableValueToObject(mojo, property,
108 					Boolean.parseBoolean(value));
109 		else
110 			setVariableValueToObject(mojo, property, value);
111 	}
112 
113 	@Given("^Server (.*)'s (.*) is (.*)$")
114 	public void setServerProperty(String serverId, String property, String value)
115 			throws IllegalAccessException {
116 		Server server = settings.getServer(serverId);
117 		if (server == null) {
118 			server = new Server();
119 			server.setId(serverId);
120 			settings.addServer(server);
121 		}
122 		setVariableValueToObject(server, property, value);
123 	}
124 
125 	@Given("^an empty Server with id (.*)$")
126 	public void addEmptyServer(String serverId) throws IllegalAccessException {
127 		Server server = settings.getServer(serverId);
128 		if (server != null)
129 			throw new AssertionFailedError(String.format(
130 					"There's already a server with ID %s", serverId));
131 		server = new Server();
132 		settings.addServer(server);
133 		setVariableValueToObject(server, "id", serverId);
134 	}
135 
136 	@Given("^the Project (.*) property is (.*)$")
137 	public void setProjectProperty(String property, String value)
138 			throws IllegalAccessException {
139 		properties.setProperty(property, value);
140 	}
141 
142 	@Given("^the System (.*) property is (.*)$")
143 	public void setSystemProperty(String property, String value)
144 			throws IllegalAccessException {
145 		systemProperties.add(property);
146 		System.setProperty(property, value);
147 	}
148 
149 	@When("^the (.*) Goal is executed")
150 	public void executeMojo(String goal) {
151 		try {
152 			mojo.execute();
153 		} catch (MojoExecutionException e) {
154 			thrown = e;
155 		} catch (MojoFailureException e) {
156 			thrown = e;
157 		}
158 	}
159 
160 	@Then("^there should be a Project (.*) property with value (.*)$")
161 	public void assertProjectProperty(String property, String value) {
162 		if ("\"\"".equals(value))
163 			value = "";
164 		assertEquals(value, properties.getProperty(property));
165 	}
166 
167 	@Then("^there should be a System (.*) property with value (.*)$")
168 	public void assertSystemProperty(String property, String value) {
169 		assertEquals(value, System.getProperty(property));
170 	}
171 
172 	@Then("^there should be an error message '(.*)'$")
173 	public void anExceptionIsThrown(String message) {
174 		assertEquals(message, thrown.getMessage());
175 	}
176 
177 	@After
178 	public void after() {
179 		for (String property : systemProperties) {
180 			System.clearProperty(property);
181 		}
182 		systemProperties.clear();
183 	}
184 }
